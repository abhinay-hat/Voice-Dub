  <behavior>
validate_asr_output(result: ASRResult) -> tuple[bool, str]:
  - Empty segments list -> (False, "No speech segments detected in video.")
  - All segments high confidence (>= ASR_CONFIDENCE_THRESHOLD 0.7) -> (True, "ASR complete: N segments, M speakers.")
  - >50% segments low confidence -> (True, "ASR complete: N segments. Warning: X/N segments have low confidence — review flagged rows.")
  - result=None -> (False, "ASR stage did not return a result.")

validate_translation_output(result: TranslationResult) -> tuple[bool, str]:
  - result=None -> (False, "Translation stage did not return a result.")
  - No segments -> (False, "No translated segments in result.")
  - >20% segments have is_valid_duration=False -> (False, "Translation failed: X/N segments exceed timing tolerance. Processing stopped.")
  - All valid -> (True, "Translation complete: N segments translated.")
  - >0 but <=20% invalid duration -> (True, "Translation complete: N segments. Warning: X segments exceed timing tolerance.")

validate_tts_output(result: TTSResult) -> tuple[bool, str]:
  - result=None -> (False, "TTS stage did not return a result.")
  - failed_segments / total_segments > 0.5 -> (False, "Voice cloning failed: >50% of segments rejected. Processing stopped.")
  - flagged_count > 0 -> (True, "Voice cloning complete: N segments. Warning: X segments flagged for review.")
  - All passed -> (True, "Voice cloning complete: N segments synthesized.")

validate_lip_sync_output(result: LipSyncResult) -> tuple[bool, str]:
  - result=None -> (False, "Lip sync stage did not return a result.")
  - result.sync_validation is not None and result.sync_validation.passed is False -> (False, f"Lip sync validation failed: {result.sync_validation.pass_rate:.1%} frames passed.")
  - result.fallback_used=True -> (True, "Lip sync complete (Wav2Lip fallback used). Review output quality.")
  - Normal success -> (True, "Lip sync complete: {result.model_used} in {result.processing_time:.1f}s.")

extract_preview_clip(video_path: str, start_seconds: float, output_path: str, duration: float = 30.0) -> str:
  - Uses ffmpeg subprocess: `ffmpeg -y -ss {start_seconds} -i {video_path} -t {duration} -c copy {output_path}`
  - Returns output_path on success
  - Raises ValueError if video_path does not exist
  - Raises RuntimeError if ffmpeg command fails (non-zero exit code)
  - Clamps start_seconds to 0 minimum (never negative)
  </behavior>
