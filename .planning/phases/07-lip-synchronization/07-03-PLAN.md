class LipSyncResult:
    output_path: Path
    model_used: str          # "latentsync" or "wav2lip"

**Progress callback points** (matching Phase 6 pattern of 8-9 checkpoints):
- 0.05: Stage started, validating inputs
- 0.10: Audio resampled to 16kHz
- 0.15: Video analyzed (duration check, chunking decision)
- 0.25: Inference started (or first chunk)
- 0.45: Inference 50% (or midpoint chunk)
- 0.70: Inference complete (or all chunks done)
- 0.85: Reassembling chunks (or copying single output)
- 0.95: Writing result JSON
- 1.0: Stage complete

**Core logic flow:**
```python
# 1. Validate inputs
if not assembled_video_path.exists():
    raise LipSyncStageFailed(f"Input video not found: {assembled_video_path}")

# 2. Create work directory for temp files
work_dir = output_dir / "lip_sync_temp"
work_dir.mkdir(parents=True, exist_ok=True)
temp_files = []

try:
    # 3. Resample audio to 16kHz (ALWAYS - never pass 48kHz to models)
    progress_callback and progress_callback(0.05)
    audio_16k = prepare_audio_for_lipsync(assembled_video_path, work_dir)
    temp_files.append(audio_16k)
    progress_callback and progress_callback(0.10)

    # 4. Check video duration for chunking decision
    from src.lip_sync.chunker import get_video_duration
    duration = get_video_duration(assembled_video_path)
    use_chunking = duration > long_video_threshold
    progress_callback and progress_callback(0.15)

    # 5. Run inference (with or without chunking)
    start_time = time.time()
    output_video = output_dir / "lip_synced_video.mp4"
    fallback_used = False
    chunks_processed = 1

    if use_chunking:
        # Split -> process each chunk -> concatenate
        chunks = split_video_into_chunks(assembled_video_path, audio_16k, work_dir, chunk_duration)
        chunks_processed = len(chunks)
        processed_chunks = []
        for i, chunk in enumerate(chunks):
            chunk_out = work_dir / f"chunk_{chunk.index:03d}_lipsync.mp4"
            progress = 0.25 + (i / chunks_processed) * 0.45
            progress_callback and progress_callback(progress)
            try:
                run_latentsync_inference(chunk.video_path, chunk.audio_path, chunk_out, inference_steps, guidance_scale, enable_deepcache)
            except RuntimeError as e:
                logger.warning(f"LatentSync failed on chunk {i}, falling back to Wav2Lip: {e}")
                run_wav2lip_inference(chunk.video_path, chunk.audio_path, chunk_out)
                fallback_used = True
            processed_chunks.append(chunk_out)
            temp_files.extend([chunk.video_path, chunk.audio_path])
        progress_callback and progress_callback(0.70)
        concatenate_video_chunks(processed_chunks, output_video, work_dir)
        temp_files.extend(processed_chunks)
        progress_callback and progress_callback(0.85)
    else:
        # Single pass inference
        progress_callback and progress_callback(0.25)
        try:
            run_latentsync_inference(assembled_video_path, audio_16k, output_video, inference_steps, guidance_scale, enable_deepcache)
        except RuntimeError as e:
            logger.warning(f"LatentSync failed, falling back to Wav2Lip: {e}")
            run_wav2lip_inference(assembled_video_path, audio_16k, output_video)
            fallback_used = True
        progress_callback and progress_callback(0.85)

    processing_time = time.time() - start_time

    # 6. Run sync validation (advisory - never fails the stage)
    from src.lip_sync.validator import validate_lip_sync_output
    sync_validation = None
    try:
        sync_validation = validate_lip_sync_output(output_video)
        if not sync_validation.passed:
            logger.warning(
                f"Lip sync validation advisory: pass_rate={sync_validation.pass_rate:.1%} "
                f"({sync_validation.valid_frames}/{sync_validation.sampled_frames} sampled frames). "
                "This is advisory only. Output video retained."
            )
        else:
            logger.info(f"Lip sync validation passed: {sync_validation.pass_rate:.1%} of sampled frames OK")
    except Exception as e:
        logger.warning(f"Sync validation failed (non-fatal): {e}")

    # 7. Build result and write JSON
    progress_callback and progress_callback(0.95)
    result = LipSyncResult(
        output_path=output_video,
        model_used="wav2lip" if fallback_used else "latentsync",
        model_version="gan" if fallback_used else "1.6",
        inference_steps=0 if fallback_used else inference_steps,
        guidance_scale=0.0 if fallback_used else guidance_scale,
        processing_time=processing_time,
        input_video_path=assembled_video_path,
        input_audio_path=audio_16k,
        chunks_processed=chunks_processed,
        fallback_used=fallback_used,
        sync_validation=sync_validation,
        multi_speaker_mode=(speakers_detected > 1),
    )

    result_json = output_dir / "lip_sync_result.json"
    with open(result_json, "w") as f:
        json.dump(result.to_dict(), f, indent=2)

    progress_callback and progress_callback(1.0)
    return result

finally:
    # Clean up temp files (audio_16k, chunks)
    for f in temp_files:
        if f.exists():
            try:
                f.unlink()
            except OSError:
                pass
```

**SyncValidation dataclass and validate_lip_sync_output() function:**

```python
"""Lightweight lip sync output validation.

Uses ffprobe + FFmpeg frame extraction to check output video quality.
Rather than running SyncNet (complex, requires model downloads), we use
average frame brightness as a proxy for "face detected in frame".
Black or near-black frames indicate face detection failure or corruption.

This is intentionally pragmatic: real SyncNet evaluation is Phase 8 territory.
"""
from __future__ import annotations

import logging
import subprocess
import tempfile
from dataclasses import dataclass
from pathlib import Path

logger = logging.getLogger(__name__)

# Frames with average pixel brightness below this threshold are considered invalid
# (likely black frame, face detection failure, or corruption)
BRIGHTNESS_THRESHOLD = 10.0  # 0-255 scale; black=0, typical face frame=80-200

# Sample every Nth frame to keep validation fast on long videos
DEFAULT_SAMPLE_INTERVAL = 30  # Check 1 frame per second at 30fps


@dataclass
class SyncValidation:
    """Result of validate_lip_sync_output()."""
    total_frames: int       # Total frame count from ffprobe
    sampled_frames: int     # Number of frames extracted for checking
    valid_frames: int       # Frames with brightness >= BRIGHTNESS_THRESHOLD
    pass_rate: float        # valid_frames / sampled_frames (0.0 to 1.0)
    passed: bool            # True if pass_rate >= 0.95

    def to_dict(self) -> dict:
        return {
            "total_frames": self.total_frames,
            "sampled_frames": self.sampled_frames,
            "valid_frames": self.valid_frames,
            "pass_rate": self.pass_rate,
            "passed": self.passed,
        }


def validate_lip_sync_output(
    video_path: Path,
    sample_interval: int = DEFAULT_SAMPLE_INTERVAL,
    pass_threshold: float = 0.95,
) -> SyncValidation:
    """
    Validate lip sync output video by sampling frames and checking brightness.

    Extracts every sample_interval-th frame via FFmpeg and computes average
    pixel brightness. Frames below BRIGHTNESS_THRESHOLD are flagged as invalid
    (proxy for face detection failure or video corruption).

    Args:
        video_path: Path to the lip-synced output MP4.
        sample_interval: Extract 1 frame every N frames. Default 30 (1/sec at 30fps).
        pass_threshold: Minimum pass_rate to set passed=True. Default 0.95 (95%).

    Returns:
        SyncValidation dataclass with frame counts and pass_rate.

    Raises:
        FileNotFoundError: If video_path does not exist.
        RuntimeError: If ffprobe fails to read the video.
    """
    if not video_path.exists():
        raise FileNotFoundError(f"Video not found for validation: {video_path}")

    # Step 1: Get total frame count via ffprobe
    probe_result = subprocess.run(
        [
            "ffprobe", "-v", "quiet",
            "-select_streams", "v:0",
            "-count_packets",
            "-show_entries", "stream=nb_read_packets",
            "-of", "default=noprint_wrappers=1:nokey=1",
            str(video_path),
        ],
        capture_output=True,
        text=True,
    )
    if probe_result.returncode != 0:
        raise RuntimeError(
            f"ffprobe failed on {video_path}: {probe_result.stderr}"
        )

    total_frames_str = probe_result.stdout.strip()
    try:
        total_frames = int(total_frames_str)
    except ValueError:
        raise RuntimeError(
            f"ffprobe returned non-integer frame count: '{total_frames_str}'"
        )

    if total_frames == 0:
        logger.warning(f"Video has 0 frames: {video_path}")
        return SyncValidation(
            total_frames=0,
            sampled_frames=0,
            valid_frames=0,
            pass_rate=0.0,
            passed=False,
        )

    # Step 2: Extract sample frames to a temp directory and check brightness
    valid_frames = 0
    sampled_frames = 0

    with tempfile.TemporaryDirectory() as tmp:
        tmp_path = Path(tmp)
        frame_pattern = str(tmp_path / "frame_%05d.png")

        # Use select filter to extract every Nth frame
        extract_result = subprocess.run(
            [
                "ffmpeg", "-y",
                "-i", str(video_path),
                "-vf", f"select=not(mod(n\\,{sample_interval}))",
                "-vsync", "vfr",
                "-pix_fmt", "rgb24",  # Consistent format for brightness check
                frame_pattern,
            ],
            capture_output=True,
            text=True,
        )
        if extract_result.returncode != 0:
            raise RuntimeError(
                f"FFmpeg frame extraction failed: {extract_result.stderr[-1000:]}"
            )

        frame_files = sorted(tmp_path.glob("frame_*.png"))
        sampled_frames = len(frame_files)

        for frame_file in frame_files:
            brightness = _get_frame_brightness(frame_file)
            if brightness >= BRIGHTNESS_THRESHOLD:
                valid_frames += 1

    pass_rate = valid_frames / sampled_frames if sampled_frames > 0 else 0.0
    passed = pass_rate >= pass_threshold

    logger.info(
        f"Sync validation: {valid_frames}/{sampled_frames} sampled frames valid "
        f"(pass_rate={pass_rate:.1%}, passed={passed}, total_frames={total_frames})"
    )

    return SyncValidation(
        total_frames=total_frames,
        sampled_frames=sampled_frames,
        valid_frames=valid_frames,
        pass_rate=pass_rate,
        passed=passed,
    )


def _get_frame_brightness(frame_path: Path) -> float:
    """
    Compute average pixel brightness of a PNG frame using ffprobe signalstats.

    Returns average brightness on 0-255 scale. Returns 0.0 on error.
    """
    result = subprocess.run(
        [
            "ffprobe", "-v", "quiet",
            "-f", "lavfi",
            "-i", f"movie={frame_path},signalstats",
            "-show_entries", "frame_tags=lavfi.signalstats.YAVG",
            "-of", "default=noprint_wrappers=1:nokey=1",
        ],
        capture_output=True,
        text=True,
    )
    if result.returncode != 0 or not result.stdout.strip():
        return 0.0
    try:
        return float(result.stdout.strip())
    except ValueError:
        return 0.0
```
